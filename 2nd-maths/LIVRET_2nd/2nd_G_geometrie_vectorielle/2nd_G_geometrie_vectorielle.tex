\begin{His}
	Michel Chasles était un mathématicien français du XIXe siècle qui a apporté d'importantes contributions dans le domaine de la géométrie vectorielle. Il est célèbre pour avoir formulé le "théorème de Chasles", qui établit que tout mouvement d'un solide rigide dans l'espace est équivalent à une combinaison de translations et de rotations. Cette découverte a été essentielle pour le développement de la géométrie vectorielle et a permis de comprendre les transformations géométriques en termes de vecteurs. Le travail de Michel Chasles a ouvert la voie à de nouvelles avancées dans la géométrie analytique et a eu un impact significatif sur le développement ultérieur des mathématiques.
\end{His}

\section{Vecteur du plan}

\begin{DefT}{Translation dans le plan}
Soit $B$ et $C$ deux points distincts du plan.

L'image d'un point $A$ par la \textbf{translation} qui transforme $B$ en $C$ est le point $A'$ tel que $ABCA'$ soit un \textbf{parallélogramme}. 
\end{DefT}

\begin{DefT}{Vecteur}
On appelle \textbf{vecteur d'origine} $A$ et d'\textbf{extrémité} $B$, l'idée du déplacement de $A$ vers $B$.  

On le note $\vec{AB}$.

Le vecteur $\vec{AB}$ est défini par :
\begin{itemize}
\item Une \textbf{direction}.
\item Un \textbf{sens}.
\item Une \textbf{longueur}, appelée \textbf{norme} du vecteur.
\end{itemize}

La translation qui transforme $A$ en $B$ est appelée translation de vecteur $\vec{AB}$.
\end{DefT}

\mini{.55\linewidth}{
\begin{DefT}{Vecteurs égaux}
On dit que deux vecteurs $\vec{AB}$ et $\vec{CD}$ sont égaux s'ils ont :
\begin{itemize}
\item La même direction.
\item Le même sens.
\item La même norme.
\end{itemize}
\end{DefT}
}{.45\linewidth}{
\begin{Pp}
\[\vec{AB}=\vec{CD}\Leftrightarrow ABCD\text{ est un parallélogramme.}\]
\end{Pp}
}

\begin{DefT}{Représentants d'un vecteur}
\mini{.65\linewidth}{
Les vecteurs $\vec{u}$ et $\vec{AB}$ ont la \textbf{même direction}, le \textbf{même sens} et la \textbf{même longueur}, ils sont \textbf{égaux}.

On dit que le vecteur $\vec{AB}$ est un \textbf{représentant} du vecteur $\vec{u}$.

\begin{Pp}
Soit $A$ un point et $\vec{u}$ un vecteur, alors il existe un \textbf{unique point} $B$ tel que $\vec{AB}$ soit un représentant de $\vec{u}$.
\end{Pp}

\section{Opérations avec les vecteurs}

\subsection{Addition de vecteurs}

\begin{Def}
Soit $\vec{u}$ et $\vec{v}$ deux vecteurs.

Si on enchaine le translation de vecteur $\vec{u}$
puis la translation de vecteur $\vec{v}$, on obtient une translation.

On appelle somme des vecteurs $\vec{u}$
et $\vec{v}$, notée $\vec{u}+\vec{v}$, le vecteur qui lui est associé.
\end{Def}

\mini{.45\linewidth}{
\begin{DefT}{Relation de M. Chasles}
Soit $A$, $B$ et $C$ trois points, alors $\vec{AB}+\vec{BC}=\vec{AC}$
\end{DefT}
}{.53\linewidth}{
\begin{PpT}{Propriétés de l'addition de vecteurs}
Pour tous vecteurs $\vec{u}$, $\vec{v}$ et $\vec{w}$ :
\begin{itemize}
\item $\vec{u}+\vec{v}=\vec{v}+\vec{u}$, l'addition est commutative.
\item $\vec{u}+(\vec{v}+\vec{w})=(\vec{u}+\vec{v})+\vec{w}$, l'addition est associative.
\end{itemize}
\end{PpT}
}

\begin{PpT}{Propriété du parallélogramme}
Soit $ABCD$ un parallélogramme, alors $\vec{AB}+\vec{AD}=\vec{AC}$.
\end{PpT}

\subsection{Soustraction de vecteurs}

\mini{.48\linewidth}{
\begin{DefT}{Le vecteur nul : $\vec{0}$}
Soit $A$ un point du plan, on appelle vecteur $\vec{AA}$ le \textbf{vecteur nul} et on le note $\vec{0}$.
\end{DefT}
}{.48\linewidth}{
\begin{Pp}
Pour tout vecteur $\vec{u}$, $\vec{u}+\vec{0}=\vec{0}+\vec{u}=\vec{u}$
\end{Pp}
}

\mini{.48\linewidth}{
\begin{DefT}{Vecteurs opposés}
On dit que deux vecteurs $\vec{u}$ et $\vec{v}$ sont \textbf{opposés} lorsque :
\begin{itemize}
\item Ils ont la \textbf{même direction}.
\item Ils ont la \textbf{même norme}.
\item Ils sont de \textbf{sens contraires}.
\end{itemize}
On le note : $\vec{u}=-\vec{v}$.
\end{DefT}
}{.48\linewidth}{
\begin{DefT}{Soustraction de vecteurs}
Soit $\vec{u}$ et $\vec{v}$ deux vecteurs.

On définit la \textbf{différence} des vecteurs $\vec{u}$ et $\vec{v}$ par : $\vec{u}-\vec{v}=\vec{u}+(-\vec{v})$.
\end{DefT}
}

\subsection{Multiplication d'un vecteur par un nombre}

\mini{.65\linewidth}{
\begin{Def}
Soit $\vec{u}$ un vecteur et $k$ un nombre réel non nul.
Le produit du vecteur $\vec{u}$ par le nombre $k$, noté $k\vec{u}$ est le vecteur $\vec{v}$ tel que :
\begin{itemize}
\item Les vecteurs $\vec{u}$ et $\vec{v}$ ont la même direction.
\item Les vecteurs $\vec{u}$ et $\vec{v}$ ont :
Le même sens si $k>0$. \hspace{1cm} Des  sens contraires si $k<0$.
\item La norme de $\vec{v}$ est égale à $|k|$ fois la norme de $\vec{u}$.
\end{itemize}
\end{Def}
}{.3\linewidth}{
\begin{Pp}
Si $M$ est le milieu du segment $[AB]$ alors :
\begin{itemize}
\item $\vec{AB}=2\vec{AM}$
\item $\vec{AM}=\frac{1}{2}\vec{AB}$
\item $\vec{AM}=\vec{MB}$
\item $\vec{AM}+\vec{BM}=\vec{0}$
\end{itemize}
\end{Pp}
}

\section{Vecteurs colinéaires}

\begin{Def}
On dit que deux vecteurs $\vec{u}$ et $\vec{v}$ sont \textbf{colinéaires} lorsqu'ils ont la \textbf{même direction}.
\end{Def}

\begin{Pp}
Soit $\vec{u}$ et $\vec{v}$ deux vecteurs.
\[ \vec{u} \text{ et } \vec{v} \text{ sont colinéaires.}\Leftrightarrow  \text{Il existe un nombre réel } k\text{ tel que }\vec{u}=k\vec{v}\]
\end{Pp}

\end{pageCours} % Fin page de cours 3
