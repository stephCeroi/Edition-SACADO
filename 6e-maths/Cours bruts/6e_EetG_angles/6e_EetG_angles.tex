\section{Définition et notation d'un angle}

\mini{.58\linewidth}{\begin{Def}
\begin{itemize}
\item Un angle est une ouverture limitée par deux demi-droites de même origine.
\item Le point d'intersection des deux demi-droites est appelé le sommet de l'angle.
\item Les deux demi-droites sont appelées les côtés de l'angle.
\end{itemize}
\end{Def}
\begin{Nt}
L'angle représenté se note $\widehat{BAC}$ ou $\widehat{CAB}$.
\end{Nt}}

\section{Unité de mesures d'angles : Le degré}

\begin{Def}
L'unité de mesure des angles est le \textbf{degré}, on le note °.
\end{Def}

\mini{.70\linewidth}{\begin{Def}
\begin{itemize}
\item Un angle aigu mesure moins de 90°.
\item Un angle droit mesure 90°.
\item Un angle obtus mesure entre 90° et 180°.
\item Un angle plat mesure 180°.
\end{itemize}
\end{Def}}
{.26\linewidth}{\qr{•}{Mesure d'angle}}

\section{Mesurer des angles}

\begin{Mt}
\mini{.70\linewidth}{
\begin{enumerate}
    \item On place le centre du rapporteur sur le sommet de l'angle.
    \item Le « 0° » du rapporteur repose sur une extrémité de l'angle : la demi-droite [AC)
    \item Les flèches du rapporteur recouvrent l'angle.
    \item La mesure de l'angle se lit sur l'autre extrémité de l'angle : la demi-droite [AB)
\end{enumerate}}

\section{Construire des angles}

\begin{Mt}
Pour construire un angle $\widehat{ABC}$ de 68°
\begin{enumerate}
    \item On commence par tracer une demi-droite [BA)
    \item On place le rapporteur en alignant le « 0° » du rapporteur avec la demi-droite [BA).
    \item On marque l'angle à 68° sur la feuille.
    \item On trace la demi-droite [BC) partant de B et passant par la marque faite à 68°.
\end{enumerate}
\qr{•}{Construire un angle de mesure donnée}
\end{Mt}
